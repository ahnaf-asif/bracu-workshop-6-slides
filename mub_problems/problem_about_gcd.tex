\begin{frame}{Problem about GCD}
	\framesubtitle{Codeforces 2043D}
	\begin{itemize}
        \item <1-> Given three integers $l$, $r$, and $G$. ($1 \leq l \leq r \leq 10^{18}$, $1 \leq G \leq 10^{18}$)
		\item <2-> Find two integers $A$ and $B$ such that:
		      \begin{align*}
			       & l \leq A \leq B \leq r         \\
			       & \gcd(A, B) = G                 \\
			       & | A - B| \text{ is maximized.}
		      \end{align*}
		      If there are multiple such pairs, output the one with the smallest $A$. \\
		      If there is no such pair, output -1 -1.
	\end{itemize}
\end{frame}

\begin{frame}{Problem about GCD}
	\framesubtitle{Codeforces 2043D}
	\begin{block}{Sample Input}
		4 12 2
	\end{block}
	\begin{block}{Sample Output}
		4 10
	\end{block}
\end{frame}

\begin{frame}{Problem about GCD}
	\framesubtitle{Codeforces 2043D}
	\textbf{Try solving if $G = 1$} \\
	\begin{itemize}
		\item <1-> We will try $A = l$ and $B = r$. If $\gcd(A, B) = 1$, we found our answer.
		\item <2-> Then we will try $A = l$ and $B = r - 1$, and then $A = l + 1$ and $B = r$.
		\item <3-> Then, we will try $(l, r - 2), (l + 1, r - 1), (l + 2, r)$ and so on.
		\item <4-> We will keep trying until we find a pair $(A, B)$ such that $\gcd(A, B) = 1$.
		\item <5-> If $A > B$, we will stop (There is no solution).
	\end{itemize}
\end{frame}

\begin{frame}{Problem about GCD}
	\framesubtitle{Codeforces 2043D}
	\begin{theorem}<1->
		The algorithm will find atleast one pair of coprime pair between $[l, l + 30)$ and $(r - 30, r]$.
	\end{theorem}

	\begin{proof}<2->
        There are 900 pairs to be considered. \\ \pause
        If a pair is not coprime, then there exists a prime number that divides both numbers. \\\pause
        There will be atmost $15 \cdot 15 = 225$ pairs that are both divisible by $2$. So we are left with $675$ pairs\\ \pause
        There will be atmost $10 \cdot 10 = 100$ pairs that are both divisible by $3$. So we are left with $575$ pairs \\ \pause
        There wil be atmost $6 \cdot 6 = 36$ pairs that are both divisible by $5$. So we are left with $539$ pairs \\ \pause
        If we keep doing this for upto prime $37$, we will still have $450$ pairs. But now the prime numbers are greater than our range. 
    \end{proof}
\end{frame}

\begin{frame}{Problem about GCD}

	\begin{theorem}<1->
		The algorithm will find atleast one pair of coprime pair between $[l, l + 30)$ and $(r - 30, r]$.
	\end{theorem}

	\begin{proof}<1->
        There are 30 numbers on the left side atmost. And each number has to be included in the pair. So one number has to be divisible by more than 15 prime numbers.  \pause
        But that would mean, that number is greater than $10^{18}$. So there won't be any such number. \\ \pause 
        Hence, there will be at least one pair of coprime numbers. \\
    \end{proof}
\end{frame}
